Example of a script showing the usage of a functionÂ¶

It could work like this:

# python3
"""
    This script imports the random module and introduces the ``if`` statement
    It shows a function definition with argument and return value
"""
import random

def main():
    print("How do I feel?")
    mood = random.random()  # mood is between 0 and 1
    emotion = express_mood(mood)
    print("I am", emotion)
    print("Good bye")

def express_mood(level):
    # translate a mood-level into a text
    if level > 0.7:
        mood_text = 'happy'
    elif level > 0.5:
        mood_text = 'fine'
    elif level > 0.2:
        mood_text = 'ok'
    else:
        mood_text = 'sad'
    return mood_text

main()
